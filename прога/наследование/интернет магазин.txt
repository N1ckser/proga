#include <iostream>
#include <string>
#include <vector>

class Товар {
public:
    std::string название;
    float цена;
    int рейтинг;
};

class Категория {
public:
    std::string имя_категории;
    std::vector<Товар> товары;
};

class Basket {
public:
    std::vector<Товар> купленные_товары;
};

class User {
public:
    std::string логин;
    std::string пароль;
    Basket корзина;
};

int main() {
    system("chcp 65001");
    Товар товар1;
    товар1.название = "Товар 1";
    товар1.цена = 100.0;
    товар1.рейтинг = 4;

    Товар товар2;
    товар2.название = "Товар 2";
    товар2.цена = 150.0;
    товар2.рейтинг = 5;

    Категория категория1;
    категория1.имя_категории = "Категория 1";
    категория1.товары.push_back(товар1);
    категория1.товары.push_back(товар2);

    User пользователь1;
    пользователь1.логин = "user1";
    пользователь1.пароль = "password1";

    пользователь1.корзина.купленные_товары.push_back(товар1);

    // Вывод на консоль каталога продуктов
    std::cout << "Каталог продуктов:\n";
    for (const auto& товар : категория1.товары) {
        std::cout << "Название: " << товар.название << ", Цена: " << товар.цена << ", Рейтинг: " << товар.рейтинг << "\n";
    }

    // Вывод на консоль покупок пользователей
    std::cout << "Покупки пользователя " << пользователь1.логин << ":\n";
    for (const auto& товар : пользователь1.корзина.купленные_товары) {
        std::cout << "Название: " << товар.название << ", Цена: " << товар.цена << ", Рейтинг: " << товар.рейтинг << "\n";
    }

    return 0;
}
